Step 2: Robust Error Checking & Parsing (parsing_safe.c)
This updated version handles pure string errors, skips empty spaces, and checks for INT Overflow / Underflow using a long receiver.

C
#include "push_swap.h"

int	stack_a_is_correct(char *str)
{
	int	i;

	i = 0;
	while (str[i] == ' ' || (str[i] >= 9 && str[i] <= 13))
		i++;
	if (str[i] == '-' || str[i] == '+')
		i++;
	if (str[i] < '0' || str[i] > '9')
		return (1); // Error: Symbol without a trailing digit or pure letter
	while (str[i])
	{
		if (str[i] < '0' || str[i] > '9')
			return (1); // Error: Embedded invalid character
		i++;
	}
	return (0);
}

long	ft_atoi_long(const char *nptr)
{
	int		i;
	long	num;
	int		neg;

	i = 0;
	num = 0;
	neg = 1;
	while (nptr[i] == ' ' || (nptr[i] >= 9 && nptr[i] <= 13))
		i++;
	if (nptr[i] == '-')
		neg = -1;
	if (nptr[i] == '-' || nptr[i] == '+')
		i++;
	while (nptr[i] >= '0' && nptr[i] <= '9')
	{
		num = (num * 10) + (nptr[i] - '0');
		i++;
	}
	return (num * neg);
}

int	*transform_argv(int argc, char **argv, int length)
{
	int		*array;
	int		i;
	int		j;
	long	val;

	array = malloc(sizeof(int) * length);
	if (!array)
		return (NULL);
	i = 1;
	j = 0;
	while (i < argc)
	{
		if (argv[i] == NULL)
		{
			i++;
			continue ;
		}
		if (stack_a_is_correct(argv[i]) == 1)
			return (free(array), NULL);
		val = ft_atoi_long(argv[i]);
		if (val < -2147483648 || val > 2147483647)
			return (free(array), NULL); // Error: INT Limit Overflow
		array[j++] = (int)val;
		i++;
	}
	return (array);
}
Step 3: Central Operations Wrapper (operations.c)
To fulfill the Benchmark (--bench) requirement, we cannot trigger standard rules directly without updating our tracker variables. We wrap them like this:

#include "push_swap.h"

// Example wrapper for sa. Do the same structural adjustment for sb, pb, ra, etc.
void	instr_sa(t_list **stack_a, t_settings *st)
{
	void	*temp;

	if (!stack_a || !*stack_a || !(*stack_a)->next)
		return ;
	temp = (*stack_a)->content;
	(*stack_a)->content = (*stack_a)->next->content;
	(*stack_a)->next->content = temp;
	
	write(1, "sa\n", 3);
	if (st)
	{
		st->stats.sa++;
		st->stats.total++;
	}
}

void	instr_pb(t_list **stack_a, t_list **stack_b, t_settings *st)
{
	t_list	*moving_node;

	if (!stack_a || !*stack_a)
		return ;
	moving_node = *stack_a;
	*stack_a = (*stack_a)->next;
	moving_node->next = *stack_b;
	*stack_b = moving_node;

	write(1, "pb\n", 3);
	if (st)
	{
		st->stats.pb++;
		st->stats.total++;
	}
}

#include "push_swap.h"

double	calculate_disorder(t_list *stack_a, int length)
{
	t_list	*curr_i;
	t_list	*curr_j;
	int		mistakes;
	int		total_pairs;

	if (length < 2)
		return (0.0);
	mistakes = 0;
	total_pairs = 0;
	curr_i = stack_a;
	while (curr_i)
	{
		curr_j = curr_i->next;
		while (curr_j)
		{
			total_pairs++;
			if ((long)curr_i->content > (long)curr_j->content)
				mistakes++;
			curr_j = curr_j->next;
		}
		curr_i = curr_i->next;
	}
	return ((double)mistakes / total_pairs);
}

void	execute_sorting(t_list **stack_a, t_list **stack_b, int len, t_settings *st)
{
	st->disorder = calculate_disorder(*stack_a, len);
	
	if (st->strategy == STRAT_SIMPLE)
		sort_simple(stack_a, stack_b, len, st);
	else if (st->strategy == STRAT_MEDIUM)
		sort_medium(stack_a, stack_b, len, st);
	else if (st->strategy == STRAT_COMPLEX)
		sort_complex(stack_a, stack_b, len, st);
	else
		sort_adaptive(stack_a, stack_b, len, st);
}

void	sort_adaptive(t_list **stack_a, t_list **stack_b, int len, t_settings *st)
{
	if (st->disorder < 0.2)
	{
		// Low Disorder: O(n) check and minimal swap optimization
		if (st->disorder > 0.0)
			instr_sa(stack_a, st);
	}
	else if (st->disorder >= 0.2 && st->disorder < 0.5)
		sort_medium(stack_a, stack_b, len, st);
	else
		sort_complex(stack_a, stack_b, len, st);
}

#include "push_swap.h"

static int	has_duplicates(int *array, int length)
{
	int	i;
	int	j;

	i = 0;
	while (i < length)
	{
		j = i + 1;
		while (j < length)
		{
			if (array[i] == array[j])
				return (1);
			j++;
		}
		i++;
	}
	return (0);
}

int	main(int argc, char **argv)
{
	t_settings	settings;
	int			*array;
	int			len;
	t_list		*stack_a;
	t_list		*stack_b;

	if (argc < 2)
		return (0);
	init_settings(&settings);
	parse_flags(argc, argv, &settings);
	len = count_numeric_args(argc, argv);
	if (len == 0)
		return (0);
	array = transform_argv(argc, argv, len);
	if (!array || has_duplicates(array, len))
	{
		if (array)
			free(array);
		write(2, "Error\n", 6);
		return (1);
	}
	stack_a = NULL;
	stack_b = NULL;
	// Convert array to stack dynamically here
	execute_sorting(&stack_a, &stack_b, len, &settings);
	print_benchmark(&settings);
	free_stack(&stack_a);
	free_stack(&stack_b);
	free(array);
	return (0);
}
